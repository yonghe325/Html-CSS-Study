object, 一种引用型数据类型
一种==无序==的数据集合(类似于c++中的类)
```js
let obj={

	uname:'张三'.
	age:18,
	gender: '男'
}
```
# 对象的组成
属性(变量) + 方法(函数)
类似于c++中的成员变量,行为
# 对象属性的操作
```js
    let obj={
      uname:'小米10青春版',
      num:'100012816024',
      weight:'0.55kg',
      address:'中国大陆'
    }
    //2. 查
    console.log(obj.address)
    console.log(obj.uname)
    console.log(obj.num)
    //3. 改
    obj.num='555'
    console.log(obj.num)
    //4. 增
    obj.price=1000
    console.log(obj.price)
    //5. 删
    console.log(obj.weight)
    delete obj.weight
    console.log(obj.weight)
```
# 对象的方法
```js
    let obj={
      uname:'小米10青春版',
      num:'100012816024',
      weight:'0.55kg',
      address:'中国大陆',
      out : function() {
        document.write(obj.uname,"</br>")
        document.write(obj.num,"</br>")
        document.write(obj.weight,"</br>")
        document.write(obj.address,"</br>")
      }
    }
    obj.out()
```
