# 开启定时器
使用 `setInterval` 调用定时器
```js
setInterval(函数,间隔时间)
    function fn() {
      console.log('两秒执行一次');
    }
    setInterval(fn,2000)
//调用已有函数
    setInterval(function(){
      console.log('一秒执行一次');
    },1000)
//匿名函数直接调用
```
+ 间隔时间的单位是==毫秒ms==
# 关闭定时器
每个定时器`setInterval`都有其独有的返回值
其返回值是一个数值型, 表示其id
```js
    function fn() {
      console.log('两秒执行一次');
    }
    let sIv_id1 =  setInterval(fn,2000)
    console.log(sIv_id1);
//输出1
    let sIv_id2 =  setInterval(fn,2000)
    console.log(sIv_i2);
//输出2
```
使用 `clearInterval` 关闭定时器
```js
	clearInterval([定时器id号(返回值)])
```
```js
    function fn() {
      console.log('两秒执行一次');
    }
    let sIv_id1 =  setInterval(fn,2000)
    console.log(sIv_id1);
//输出1
    let sIv_id2 =  setInterval(fn,2000)
    console.log(sIv_i2);
//输出2
	clearInterval(1)
//关闭返回值为 1 的定时器

```
通过绑定事件关闭定时器
