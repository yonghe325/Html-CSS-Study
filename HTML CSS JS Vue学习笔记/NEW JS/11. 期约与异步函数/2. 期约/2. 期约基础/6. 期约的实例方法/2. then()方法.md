`then()`方法 是 为 期约 实例添加处理程序的主要方法
`then()`方法最多接收两个参数, 分别是 `onResolved` 处理程序和 `onRejected` 处理程序 作为回调函数
	且这两个参数只能是函数对象, 否则静默忽略
这两个参数都是可选的，如果提供的话，则会在期约分别进入 “兑现” 和 “拒绝” 状态时执行
	即当期约切换状态是, 根据切换后的状态, 执行对应的代码
```js
function onResolved(id) {
	setTimeout(console.log, 0, id, 'resolved');
}
function onRejected(id) {
	setTimeout(console.log, 0, id, 'rejected');
}
	//状态处理程序

let p1 = new Promise((resolve, reject) => setTimeout(resolve, 3000));
let p2 = new Promise((resolve, reject) => setTimeout(reject, 3000));
	//实例化期约
p1.then(() => onResolved('p1'),() => onRejected('p1'));
p2.then(() => onResolved('p2'),() => onRejected('p2'));
	//为期约的then方法传入两个参数
	//分别指定对应期约进入 resolve状态 或 reject状态
//（3 秒后）
// p1 resolved
// p2 rejected
```
如果不想指定 对应的状态处理程序, 可以传入`null`
```js
function onResolved(id) {
	setTimeout(console.log, 0, id, 'resolved');
}
function onRejected(id) {
	setTimeout(console.log, 0, id, 'rejected');
}
let p1 = new Promise((resolve, reject) => setTimeout(resolve, 3000));
let p2 = new Promise((resolve, reject) => setTimeout(reject, 3000));

// 非函数处理程序会被静默忽略，不推荐
p1.then('gobbeltygook');

// 不传 onResolved 处理程序的规范写法
p2.then(null, () => onRejected('p2'));
// p2 rejected（3 秒后）
```
==状态处理程序仅有 唯一参数==
即为对应期约的 值(value) 或 是 错误信息(reason)
# Promise类型then()的返回值
`Promise.prototype.then()`方法始终返回一个新的期约实例
新 `Promise` 的状态和值取决于
+ `then` 执行的是`onResolved` 处理程序 还是 `onRejected` 处理程序
+ 回调函数的执行结果
## 执行`onResolved` 处理程序时
```js
let p1 = Promise.resolve('foo');
```
### 不传入回调函数
返回一个状态和值相同的新期约实例
```js
let p2 = p1.then();
setTimeout(console.log, 0, p2); // Promise <resolved>: foo
```
### 回调函数 无显式返回值 或 返回`Undefined`
返回值将被`Promise.resolve()`包装, 值为`Undefined`
	转到[[3. 通过执行函数控制期约状态#`Promise.resolve()`|Promise.resolve()包装]]
```js
let p3 = p1.then(() => undefined);
	//返回Undefined
let p4 = p1.then(() => {});
let p5 = p1.then(() => Promise.resolve());
setTimeout(console.log, 0, p3); // Promise <resolved>: undefined
setTimeout(console.log, 0, p4); // Promise <resolved>: undefined
setTimeout(console.log, 0, p5); // Promise <resolved>: undefined
```
在`let p4 = p1.then(() => {});`中
`{}`被箭头函数解析为函数体, 而不是一个空对象
如果一个箭头函数无函数体, 仅返回一个空对象
应写作:
```js
let p4 = p1.then(() => ({}));
```
此时就被解析为返回一个普通值
### 回调函数返回普通值
即返回非`Promise`对象, 非`Undefined`的返回值
此时返回值将被`Promise.resolve()`包装, 值为回调函数的返回值
```js
let p6 = p1.then(() => 'bar');
setTimeout(console.log, 0, p6); // Promise <resolved>: bar
```
### 回调函数返回`Promise`对象
### 回调函数返回 `pending Promise`
返回一个状态为`pending`的新期约
```js
let p8 = p1.then(() => new Promise(() => {}));
setTimeout(console.log, 0, p8); // Promise <pending>
```
### 回调函数返回 `resolved Promise`
`then`的返回值直接跟随回调函数的返回值
```js
let p7 = p1.then(() => Promise.resolve('bar'));
setTimeout(console.log, 0, p7); // Promise <resolved>: bar
```
### 回调函数返回 `rejected Promise`
返回一个状态为`rejected`的新期约, 错误原因为`undefined`
```js
let p9 = p1.then(() => Promise.reject());
// Uncaught (in promise): undefined
setTimeout(console.log, 0, p9); // Promise <rejected>: undefined
```
### 回调函数抛出异常
返回一个状态为`rejected`的新期约
```js
let p10 = p1.then(() => { throw 'baz'; });
// Uncaught (in promise) baz
setTimeout(console.log, 0, p10); // Promise <rejected> baz
```
回调函数内部的`throw`行为
不需要`try/catch`捕获, 而是会被`Promise`内部捕获
转换为一个 `rejected Promise`, 不会像普通错误那样中断程序
同时抛出的异常的描述
会被作为新期约的 错误原因
### 回调函数返回错误值
如果回调函数内部返回错误值
会把==错误对象会被`Promise.resolve()`包装==
同时把错误对象作为值存储
```js
let p11 = p1.then(() => Error('qux'));
setTimeout(console.log, 0, p11); // Promise <resolved>: Error: qux
```
## 执行`onRejected` 处理程序时
`onRejected` 处理程序返回的值也会被 `Promise.resolve()`包装
`onRejected` 处理程序的任务不就是捕获异步错误吗
因此, 拒绝处理程序在捕获异步错误后不抛出异常是符合期约的行为
所以应该返回一个解决期约
```js
let p1 = Promise.reject('foo');
// 调用 then()时不传处理程序则原样向后传
let p2 = p1.then();
// Uncaught (in promise) foo
setTimeout(console.log, 0, p2); // Promise <rejected>: foo

// 这些都一样->无返回值/无无显式返回值
let p3 = p1.then(null, () => undefined);
let p4 = p1.then(null, () => {});
let p5 = p1.then(null, () => Promise.resolve());
setTimeout(console.log, 0, p3); // Promise <resolved>: undefined
setTimeout(console.log, 0, p4); // Promise <resolved>: undefined
setTimeout(console.log, 0, p5); // Promise <resolved>: undefined

// 这些都一样->返回普通值/返回resolved期约对象
let p6 = p1.then(null, () => 'bar');
let p7 = p1.then(null, () => Promise.resolve('bar'));
setTimeout(console.log, 0, p6); // Promise <resolved>: bar
setTimeout(console.log, 0, p7); // Promise <resolved>: bar

// Promise.resolve()保留返回的期约 -> 原样返回
let p8 = p1.then(null, () => new Promise(() => {}));
let p9 = p1.then(null, () => Promise.reject());
// Uncaught (in promise): undefined
setTimeout(console.log, 0, p8); // Promise <pending>
setTimeout(console.log, 0, p9); // Promise <rejected>: undefined

//抛出异常->同样是返回rejected状态的期约
let p10 = p1.then(null, () => { throw 'baz'; });
// Uncaught (in promise) baz
setTimeout(console.log, 0, p10); // Promise <rejected>: baz

//返回错误->同样返回resolved状态的期约
let p11 = p1.then(null, () => Error('qux'));
setTimeout(console.log, 0, p11); // Promise <resolved>: Error: qux
```
