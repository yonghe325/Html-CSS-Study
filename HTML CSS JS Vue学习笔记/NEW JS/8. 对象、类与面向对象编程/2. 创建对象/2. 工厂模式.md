按照特定接口直接创建对象, 被称为工厂模式
```js
function createPerson(name, age, job) {
	let o = new Object();
	o.name = name;
	o.age = age;
	o.job = job;
	o.sayName = function() {
		console.log(this.name);
	};
return o;
}
let person1 = createPerson("Nicholas", 29, "Software Engineer");
let person2 = createPerson("Greg", 27, "Doctor");
```
---
以工厂模式创建对象, 无法判断[[对象标识类型]](即新创建的对象是什么类型)
仅能解决创建多个类似对象
```js
function createPerson(name, age) {
    return {
        name: name,
        age: age,
        sayHello() {
            console.log(`Hello, I'm ${this.name}`);
        }
    };
}

function createAnimal(name) {
    return {
        name: name,
        eat() {
            console.log(`${this.name} is eating`);
        }
    };
}

const alice = createPerson('Alice', 25);
const rex = createAnimal('Rex');

// ❌ 无法区分对象类型
console.log(alice instanceof createPerson);  // false
console.log(rex instanceof createAnimal);    // false

// ❌ 所有对象都是 Object 类型
console.log(alice instanceof Object);  // true
console.log(rex instanceof Object);    // true

// ❌ 无法判断 alice 是 Person 还是 Animal
// 因为两者结构相似，无法区分
```
